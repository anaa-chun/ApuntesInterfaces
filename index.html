PRÁCTICA: control de un potenciómetro

Vamos a desarrollar el ejemplo del potenciómetro siguiendo la actividad propuesta. Este control visualizará un valor entre un mínimo y un máximo, iluminando una serie de elementos luminosos (seis verdes, tres amarillos y uno rojo) para representar gráficamente el progreso.

Código del Potenciómetro

using System;
using System.ComponentModel;
using System.Drawing;
using System.Windows.Forms;

public class Potenciometro : UserControl
{
    private int min = 0;
    private int max = 100;
    private int valor = 0;
    private const int elementos = 10; // Número de luces

    [Description("Valor mínimo del potenciómetro"), DefaultValue(0)]
    public int Min
    {
        get { return min; }
        set
        {
            if (value < max)
            {
                min = value;
                Invalidate();
            }
        }
    }

    [Description("Valor máximo del potenciómetro"), DefaultValue(100)]
    public int Max
    {
        get { return max; }
        set
        {
            if (value > min)
            {
                max = value;
                Invalidate();
            }
        }
    }

    [Description("Valor actual del potenciómetro"), DefaultValue(0)]
    public int Valor
    {
        get { return valor; }
        set
        {
            if (value >= min && value <= max)
            {
                valor = value;
                Invalidate();
            }
        }
    }

    public Potenciometro()
    {
        this.Width = 50;
        this.Height = 200;
        this.DoubleBuffered = true;
    }

    protected override void OnPaint(PaintEventArgs e)
    {
        base.OnPaint(e);
        Graphics g = e.Graphics;

        int ancho = this.Width - 10; // Margen lateral
        int altoElemento = (this.Height - 20) / elementos; // Altura de cada luz
        int margen = 5; // Margen superior e inferior

        // Calcular cuántos elementos encender
        int encendidos = (int)Math.Round(((double)(valor - min) / (max - min)) * elementos);

        for (int i = 0; i < elementos; i++)
        {
            Brush color = Brushes.Gray; // Por defecto, luz apagada

            // Determinar el color según la posición
            if (i < encendidos)
            {
                if (i < 6) color = Brushes.Green; // Primeros 6: verde
                else if (i < 9) color = Brushes.Yellow; // Siguientes 3: amarillo
                else color = Brushes.Red; // Último: rojo
            }

            // Dibujar el rectángulo de la luz
            g.FillRectangle(color, margen, margen + i * altoElemento, ancho, altoElemento - 2);
        }
    }
}

Explicación del Código

  1. Propiedades Personalizadas:
    • Min: Establece el valor mínimo del rango.
    • Max: Establece el valor máximo del rango.
    • Valor: Representa el valor actual, con validación para que esté dentro de los límites (min y max).
  1. Cálculo del Progreso:
    • Determina cuántos elementos deben estar encendidos basándose en la relación del valor actual con el rango (min a max).
  1. Dibujado del Control:
    • Cada elemento se representa como un rectángulo. Según su posición, se le asigna un color:
      • Verde: Los primeros 6.
      • Amarillo: Los siguientes 3.
      • Rojo: El último.
  1. Invalidate:
    • Asegura que el control se redibuje cada vez que cambian las propiedades (Min, Max o Valor).

Funcionalidades Adicionales

  1. Evento CambioValor: Se activará cuando el valor del potenciómetro cambie, notificando a los suscriptores.
  1. Propiedad Incremento: Define en cuánto se incrementa o decrementa el valor cuando el usuario interactúa con el control.
  1. Interacción con el ratón: Al hacer clic en las luces, se ajustará el valor del potenciómetro.

Código del Potenciómetro Mejorado

using System;
using System.ComponentModel;
using System.Drawing;
using System.Windows.Forms;

public class Potenciometro : UserControl
{
    private int min = 0;
    private int max = 100;
    private int valor = 0;
    private int incremento = 10;
    private const int elementos = 10;

    [Description("Valor mínimo del potenciómetro"), DefaultValue(0)]
    public int Min
    {
        get { return min; }
        set
        {
            if (value < max)
            {
                min = value;
                if (valor < min) Valor = min;
                Invalidate();
            }
        }
    }

    [Description("Valor máximo del potenciómetro"), DefaultValue(100)]
    public int Max
    {
        get { return max; }
        set
        {
            if (value > min)
            {
                max = value;
                if (valor > max) Valor = max;
                Invalidate();
            }
        }
    }

    [Description("Valor actual del potenciómetro"), DefaultValue(0)]
    public int Valor
    {
        get { return valor; }
        set
        {
            if (value >= min && value <= max)
            {
                valor = value;
                Invalidate();
                OnCambioValor(new EventArgs());
            }
        }
    }

    [Description("Incremento al interactuar con el control"), DefaultValue(10)]
    public int Incremento
    {
        get { return incremento; }
        set
        {
            if (value > 0) incremento = value;
        }
    }

    // Evento personalizado
    public event EventHandler CambioValor;

    protected virtual void OnCambioValor(EventArgs e)
    {
        CambioValor?.Invoke(this, e);
    }

    public Potenciometro()
    {
        this.Width = 50;
        this.Height = 200;
        this.DoubleBuffered = true;

        this.MouseClick += Potenciometro_MouseClick;
    }

    protected override void OnPaint(PaintEventArgs e)
    {
        base.OnPaint(e);
        Graphics g = e.Graphics;

        int ancho = this.Width - 10;
        int altoElemento = (this.Height - 20) / elementos;
        int margen = 5;

        int encendidos = (int)Math.Round(((double)(valor - min) / (max - min)) * elementos);

        for (int i = 0; i < elementos; i++)
        {
            Brush color = Brushes.Gray;

            if (i < encendidos)
            {
                if (i < 6) color = Brushes.Green;
                else if (i < 9) color = Brushes.Yellow;
                else color = Brushes.Red;
            }

            g.FillRectangle(color, margen, margen + i * altoElemento, ancho, altoElemento - 2);
        }
    }

    private void Potenciometro_MouseClick(object sender, MouseEventArgs e)
    {
        int altoElemento = (this.Height - 20) / elementos;
        int indice = (e.Y - 5) / altoElemento; // Calcular el índice del elemento clicado

        if (indice >= 0 && indice < elementos)
        {
            // Ajustar el valor según el índice
            Valor = min + (indice + 1) * (max - min) / elementos;
        }
    }
}

Explicación de las Funcionalidades Nuevas

  1. Evento CambioValor:
    • Se lanza cada vez que el valor cambia, ya sea por interacción del usuario o mediante código.
    • Este evento permite que otros componentes en la aplicación reaccionen automáticamente al cambio de valor.
  1. Propiedad Incremento:
    • Define el tamaño del paso para aumentar o disminuir el valor.
    • Aunque en este ejemplo no se utiliza explícitamente, se podría combinar con teclas del teclado o controles adicionales.
  1. Interacción del Ratón:
    • Al hacer clic en una luz, el control ajusta el valor del potenciómetro proporcionalmente.
    • Esto permite al usuario interactuar directamente con el componente.

1. Soporte para el teclado

Permitiremos ajustar el valor con las teclas flecha arriba y flecha abajo, usando la propiedad Incremento para determinar cuánto aumenta o disminuye.

Implementación:

Añadimos el manejo del teclado sobrescribiendo el evento OnKeyDown:

protected override void OnKeyDown(KeyEventArgs e)
{
    base.OnKeyDown(e);

    if (e.KeyCode == Keys.Up)
    {
        // Incrementar valor
        Valor = Math.Min(Valor + Incremento, Max);
    }
    else if (e.KeyCode == Keys.Down)
    {
        // Decrementar valor
        Valor = Math.Max(Valor - Incremento, Min);
    }
}

Detalle:

  • Keys.Up y Keys.Down: Detecta las flechas.
  • Math.Min y Math.Max: Garantizan que el valor permanezca dentro del rango [Min, Max].

Para asegurarte de que el control reciba los eventos del teclado, configura su propiedad TabStop a true y usa this.Focus() si es necesario.

3. Transiciones suaves

Para mejorar la estética, implementaremos una animación que encienda o apague las luces gradualmente.

Implementación:

Usaremos un Timer que actualice progresivamente el valor actual hasta alcanzar el objetivo.

private Timer animacionTimer;
private int objetivoValor;

public Potenciometro()
{
    animacionTimer = new Timer { Interval = 50 }; // Intervalo de la animación
    animacionTimer.Tick += AnimacionTimer_Tick;
}

private void AnimacionTimer_Tick(object sender, EventArgs e)
{
    if (valor < objetivoValor)
    {
        Valor = Math.Min(valor + 1, objetivoValor);
    }
    else if (valor > objetivoValor)
    {
        Valor = Math.Max(valor - 1, objetivoValor);
    }
    else
    {
        animacionTimer.Stop(); // Detener animación al llegar al valor objetivo
    }
}

public void CambiarValorConAnimacion(int nuevoValor)
{
    if (nuevoValor >= Min && nuevoValor <= Max)
    {
        objetivoValor = nuevoValor;
        animacionTimer.Start();
    }
}
